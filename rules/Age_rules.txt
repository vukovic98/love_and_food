rule "Age_18_20"
	salience 100
	agenda-group "age-group"
	when
		$user: User( age==Age.Age_18_20, $id: id );
		$partnerCharacteristics: PartnerCharacteristics(userId==$id);
	then
		modify($partnerCharacteristics){setIdealAgeGroup(Age.Age_18_20); 
						getSuitableAgeGroup().add(Age.Age_20_25); };
end;
