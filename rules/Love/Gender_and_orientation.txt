rule "M + Straight"
	salience 100
	agenda-group "preferred-gender"
	when
		$user: User( gender==Gender.MALE && orientetion==Orientation.HETERO, $id: id );
		$partnerCharacteristics: PartnerCharacteristics(userId==$id);
	then
		modify($partnerCharacteristics){setPreferredGender(Gender.FEMALE);};
end;
